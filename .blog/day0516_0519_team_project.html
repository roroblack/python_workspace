CH 02 DAY7DB 스키마 — regions FK 중심의 5 테이블 + SQLAlchemy#

학습 — 5 테이블 스키마 (실제 dbscript_table.sql)

-- crawler_db : utf8mb4_unicode_ci
CREATE TABLE regions (
    region_id   SMALLINT    PRIMARY KEY AUTO_INCREMENT,
    region_name VARCHAR(20) NOT NULL UNIQUE   -- '서울', '경기' ... 17 시도
);

CREATE TABLE car_registrations (
    id          BIGINT    PRIMARY KEY AUTO_INCREMENT,
    region_id   SMALLINT  NOT NULL,
    stat_year   SMALLINT  NOT NULL,
    count       INT       DEFAULT 0,
    FOREIGN KEY (region_id) REFERENCES regions(region_id),
    UNIQUE KEY uq_stat (region_id, stat_year)   -- 연도×지역 중복 방지
);

CREATE TABLE hydrogen_charging_station (
    id           INT         PRIMARY KEY AUTO_INCREMENT,
    region_id    SMALLINT    NOT NULL,
    station_name VARCHAR(100) NOT NULL,
    address      VARCHAR(255),
    lat          DECIMAL(10,7),
    lon          DECIMAL(10,7),
    FOREIGN KEY (region_id) REFERENCES regions(region_id)
);

CREATE TABLE faq (
    faq_id   INT  PRIMARY KEY AUTO_INCREMENT,
    question TEXT NOT NULL,
    answer   TEXT
);

CREATE TABLE crawl_stat (                  -- 크롤링 이력
    crawl_id      INT         PRIMARY KEY AUTO_INCREMENT,
    target_type   VARCHAR(30) NOT NULL,
    last_crawled_at DATETIME,
    CHECK (target_type IN ('car_registration','station','faq'))
);

학습 — SQLAlchemy + PyMySQL 연결 레이어

# data/db.py — .env 에서만 읽고, 하드코딩 없음
from sqlalchemy import create_engine
from sqlalchemy.engine import URL
from dotenv import load_dotenv

def get_engine():
    db_url = URL.create(
        "mysql+pymysql",
        username=_require_env("DB_USER"),
        password=_require_env("DB_PASSWORD"),
        host=_require_env("DB_HOST"),
        port=int(_require_env("DB_PORT")),
        database=_require_env("DB_NAME"),
        query={"charset": "utf8mb4"},
    )
    return create_engine(db_url, pool_pre_ping=True)
pool_pre_ping: When set to True, will enable the "pre-ping" feature that tests connections for liveness upon each checkout. — SQLAlchemy — Engine Configuration

의문 → 가설

가설 regions 테이블을 별도로 두고 FK 로 연결하는 이유는 — "서울특별시"/"서울시"/"서울" 같이 출처마다 달라지는 시도명을 정규화된 하나의 형태로 모아두기 위해서다. 크롤러가 뭐라고 불러와도 regions.region_name 에 정규화한 뒤 그 region_id를 저장하면, UI 의 지역 필터는 어느 테이블에서 조회하든 같은 region_id 로 작동한다.

결론(중간)

KEY POINT 차원 테이블(regions) + 팩트 테이블(car_registrations, hydrogen_charging_station) 의 스타 스키마 구조 — 시도명 정규화를 DB 레벨에서 강제하기 때문에, UI 와 크롤러가 서로 다른 이름을 써도 쿼리는 같은 결과를 돌려준다. pool_pre_ping=True 는 장시간 연결 후 재조회 시 오래된 커넥션이 끊겨있는 문제를 사전 차단한다.

# crawler_molit.py (요약)
# MolitCarCrawler — 연말 스냅샷(12월)만 수집해 속도 최적화
class MolitCarCrawler:
    TARGET_FUELS = {"수소", "수소전기"}   # 10.연료별_등록현황 시트

    def crawl(self, years_back=8) -> list[CarRegistrationItem]:
        html    = self._get_html()            # Playwright 로 목록 페이지 가져오기
        links   = self._parse_links(html)     # onclick 에서 다운로드 링크 추출

        all_items = []
        # 현재 연도: 최신 월부터 역순 탐색 (공표 시차 대응)
        for m in range(datetime.now().month, 0, -1):
            link = self._find_link(links, now.year, m)
            xlsx_bytes = self._get_file(link["onclick"])
            items = self._parse(xlsx_bytes, now.year, fuel_filter)
            if items: break   # 현재 연도 1개 월만

        # 과거 years_back년 → 12월(연말 스냅샷)만
        for i in range(1, years_back + 1):
            link = self._find_link(links, now.year - i, 12)
            items = self._parse(self._get_file(link["onclick"]), ...)
            all_items.extend(items)

        return all_items

# crawler_faq.py — ev.or.kr 의 봇 보안 챌린지 우회
class EvFaqCrawler:
    def _fetch_all_pages(self) -> list[str]:
        with sync_playwright() as p:
            browser = p.chromium.launch(
                headless=True,
                args=["--disable-blink-features=AutomationControlled", "--no-sandbox"]
            )
            ctx = browser.new_context(
                user_agent=USER_AGENT,
                viewport={"width":1280, "height":800},
                locale="ko-KR",   # 챌린지 통과 조건
            )
            page = ctx.new_page()
            page.goto(self.url, wait_until="domcontentloaded", timeout=60_000)

            # pnp4web.js 봇 챌린지 흐름:
            #   1) 챌린지 페이지 → 첫 번째 networkidle
            #   2) 챌린지 JS 실행 → 실제 FAQ 페이지로 리다이렉트
            #   3) FAQ 페이지 → 두 번째 networkidle
            page.wait_for_load_state("networkidle", timeout=30_000)  # 챌린지
            page.wait_for_timeout(3_000)                              # 리다이렉트 대기
            page.wait_for_load_state("networkidle", timeout=30_000)  # FAQ 안정화

의문 → 가설

가설 세 크롤러 모두 Playwright 를 쓰는 이유는 — 세 출처가 모두 JS 렌더링 또는 다운로드 버튼 클릭이 필요한 동적 페이지이기 때문이다. 정적 크롤링(urllib+bs4)이 아닌 Playwright 를 선택한 것은 "HTML 응답에 데이터가 없다" 는 같은 문제를 다른 형태로 마주쳤기 때문. 단, 세 크롤러의 최종 파싱 도구는 다르다 — Excel = openpyxl, CSV = pandas, HTML = bs4.

결론(중간)

KEY POINT "크롤러를 어떻게 짜야 하는가" 의 답은 출처마다 다르다 — 브라우저가 필요한 이유는 같더라도, 받아온 결과물이 Excel/CSV/HTML 중 무엇인지에 따라 파싱 코드가 완전히 달라진다. 세 크롤러를 같은 crawl() 인터페이스 뒤에 숨기는 CrawlService 가 그 차이를 흡수한다.

# app.py — APScheduler 를 @st.cache_resource 로 단일 인스턴스 보장
@st.cache_resource
def _get_station_scheduler() -> SchedulerService:
    class _StationCrawler:
        def crawl(self):
            return StationCrawler().crawl()  # crawl()이 int 반환 → DB 저장 직접 처리

    _svc = CrawlService(_StationCrawler())
    _sch = SchedulerService(_svc)
    _sch.add_interval_job(minutes=5)    # 5분마다 수소충전소 갱신
    _sch.start()
    _sch.pause()   # 초기엔 정지 상태 — 사이드바에서 사용자가 켠다
    return _sch

@st.cache_resource
def _get_faq_scheduler() -> SchedulerService:
    class _AllFaqCrawler:
        def crawl(self):
            return crawl_all_and_save()     # ev.or.kr + hyundai.com 동시 수집
    _sch = SchedulerService(CrawlService(_AllFaqCrawler()))
    _sch.add_interval_job(minutes=1440) # 24시간 주기
    _sch.start()
    _sch.pause()
    return _sch

의문 → 가설

가설 @st.cache_resource 를 스케줄러에 쓰는 이유는 — Streamlit 은 버튼 클릭·페이지 이동마다 app.py 전체를 재실행한다. 그때마다 새 스케줄러가 만들어지면 같은 잡이 중복으로 돌아간다. cache_resource 는 서버가 살아있는 동안 단 하나의 인스턴스를 반환해 이 문제를 막는다.

결론(중간)

KEY POINT Streamlit 의 "리런 모델" — 인터랙션마다 스크립트 재실행 — 은 스케줄러·DB 엔진 같은 "한 번만 만들어야 하는 객체" 와 충돌한다. @st.cache_resource 가 그 충돌을 막는 표준 패턴이다. 초기 pause 상태로 생성하는 것은 "앱 시작 즉시 크롤링" 의 의도치 않은 부하를 막기 위함이다.

# crawler_station.py — Streamlit 환경에서 Playwright 안전 실행
from concurrent.futures import ThreadPoolExecutor

def _get_csv(self) -> tuple[bytes, str]:
    # Streamlit/tornado 의 SelectorEventLoop 와 충돌 → 별도 스레드에서 실행
    with ThreadPoolExecutor(max_workers=1) as executor:
        future = executor.submit(self._run_playwright)
        return future.result()

def _run_playwright(self) -> tuple[bytes, str]:
    # Windows 에서 subprocess 를 지원하는 ProactorEventLoop 명시 생성
    loop = asyncio.ProactorEventLoop()
    return loop.run_until_complete(self._async_download())
